function confirmPayment() { const energy = document.getElementById('energy').value; const confirmBtn = document.getElementById('confirmBtn'); const successDiv = document.getElementById('successMessage'); const errorDiv = document.getElementById('errorMessage'); // Disable button to prevent double-click confirmBtn.disabled = true; confirmBtn.textContent = "⏳ Sending command..."; // Hide previous messages successDiv.style.display = 'none'; errorDiv.style.display = 'none'; // MQTT connection details const broker = "broker.hivemq.com"; const port = 8000; // WebSocket port const clientId = "client-" + Math.random().toString(36).substring(7); const client = new Paho.MQTT.Client(broker, port, clientId); // Set callback for connection loss (optional) client.onConnectionLost = function(response) { console.log("Connection lost: " + response.errorMessage); errorDiv.style.display = 'block'; confirmBtn.disabled = false; confirmBtn.textContent = "✅ I Have Sent the Money"; }; // Set callback for message arrival (not needed here) client.onMessageArrived = function(message) {}; // Connect to broker client.connect({ onSuccess: function() { console.log("Connected to MQTT broker"); // Create and send message const message = new Paho.MQTT.Message("AUTH," + energy); message.destinationName = "nodeb/commands"; client.send(message); console.log("Command sent: AUTH," + energy); // Disconnect (optional) client.disconnect(); // Show success successDiv.style.display = 'block'; successDiv.innerHTML = `✅ Command sent! Charging started with ${energy} kWh credit.`; confirmBtn.disabled = false; confirmBtn.textContent = "✅ I Have Sent the Money"; }, onFailure: function(err) { console.error("MQTT connection failed:", err); errorDiv.style.display = 'block'; confirmBtn.disabled = false; confirmBtn.textContent = "✅ I Have Sent the Money"; } }); }
